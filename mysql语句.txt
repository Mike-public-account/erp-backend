一、权限模块（RBAC）
sql
-- 用户表 sys_user
CREATE TABLE `sys_user` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '用户ID',
  `username` varchar(50) NOT NULL COMMENT '登录用户名',
  `password` varchar(128) NOT NULL COMMENT 'MD5加盐密码',
  `salt` varchar(32) NOT NULL COMMENT '加密盐值',
  `real_name` varchar(50) DEFAULT NULL COMMENT '真实姓名',
  `phone` varchar(20) DEFAULT NULL COMMENT '手机号',
  `email` varchar(100) DEFAULT NULL COMMENT '邮箱',
  `status` tinyint NOT NULL DEFAULT 1 COMMENT '0禁用 1启用',
  `last_login_time` datetime DEFAULT NULL COMMENT '最后登录时间',
  `last_login_ip` varchar(50) DEFAULT NULL COMMENT '最后登录IP',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` tinyint NOT NULL DEFAULT 0 COMMENT '逻辑删除 0未删 1已删',
  PRIMARY KEY (`id`),
  UNIQUE KEY uk_username (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统用户表';

-- 角色表 sys_role
CREATE TABLE `sys_role` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '角色ID',
  `role_code` varchar(50) NOT NULL COMMENT '角色编码 ROLE_ADMIN',
  `role_name` varchar(100) NOT NULL COMMENT '角色名称',
  `description` varchar(255) DEFAULT NULL COMMENT '角色描述',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `is_deleted` tinyint NOT NULL DEFAULT 0 COMMENT '逻辑删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY uk_role_code (`role_code`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='角色表';

-- 权限菜单表 sys_permission
CREATE TABLE `sys_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '权限ID',
  `parent_id` bigint NOT NULL DEFAULT 0 COMMENT '父ID 0根节点',
  `perm_code` varchar(100) NOT NULL COMMENT '权限标识 purchase:order:add',
  `perm_name` varchar(100) DEFAULT NULL COMMENT '权限名称',
  `perm_type` tinyint DEFAULT NULL COMMENT '1菜单 2按钮 3接口',
  `path` varchar(200) DEFAULT NULL COMMENT '前端路由',
  `icon` varchar(100) DEFAULT NULL COMMENT '菜单图标',
  `sort_order` int NOT NULL DEFAULT 0 COMMENT '排序号',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY uk_perm_code (`perm_code`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='权限菜单表';

-- 用户角色关联 sys_user_role
CREATE TABLE `sys_user_role` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `user_id` bigint NOT NULL,
  `role_id` bigint NOT NULL,
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY uk_user_role (`user_id`,`role_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户角色中间表';

-- 角色权限关联 sys_role_permission
CREATE TABLE `sys_role_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `role_id` bigint NOT NULL,
  `permission_id` bigint NOT NULL,
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY uk_role_perm (`role_id`,`permission_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='角色权限中间表';

-- 操作日志 sys_operation_log
CREATE TABLE `sys_operation_log` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '日志ID',
  `user_id` bigint DEFAULT NULL COMMENT '操作人ID',
  `username` varchar(50) DEFAULT NULL COMMENT '操作用户名',
  `module` varchar(50) DEFAULT NULL COMMENT '模块名称',
  `operation` varchar(100) DEFAULT NULL COMMENT '操作描述',
  `method` varchar(200) DEFAULT NULL COMMENT '方法全类名',
  `request_url` varchar(300) DEFAULT NULL COMMENT '请求地址',
  `request_method` varchar(10) DEFAULT NULL COMMENT 'GET/POST',
  `request_params` text DEFAULT NULL COMMENT '请求参数JSON',
  `response_code` int DEFAULT NULL COMMENT '响应码',
  `elapsed_time` int DEFAULT NULL COMMENT '耗时ms',
  `ip_address` varchar(50) DEFAULT NULL COMMENT '客户端IP',
  `error_msg` text DEFAULT NULL COMMENT '异常信息',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY idx_create_time (`create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统操作日志';

-- 库存流水 inv_stock_record（核心）
CREATE TABLE `inv_stock_record` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '流水ID',
  `material_id` bigint NOT NULL COMMENT '物料ID',
  `warehouse_id` bigint NOT NULL COMMENT '仓库ID',
  `record_type` tinyint NOT NULL COMMENT '1采购入库 2生产领料 3完工入库 4销售出库 5预占 6预出库 7释放 8盘点',
  `stock_status` tinyint NOT NULL COMMENT '1预占 2已入库 3预出库 4已出库',
  `qty_change` decimal(15,4) NOT NULL COMMENT '变动数量 正数入库',
  `unit_cost` decimal(15,6) DEFAULT NULL COMMENT '本次单位成本',
  `total_cost` decimal(15,4) DEFAULT NULL COMMENT '总成本',
  `qty_after` decimal(15,4) DEFAULT NULL COMMENT '变动后库存',
  `avg_cost_after` decimal(15,6) DEFAULT NULL COMMENT '新加权成本',
  `ref_type` varchar(50) DEFAULT NULL COMMENT '单据类型 PUR_ORDER',
  `ref_id` bigint DEFAULT NULL COMMENT '单据主ID',
  `ref_item_id` bigint DEFAULT NULL COMMENT '单据明细ID',
  `operator_id` bigint DEFAULT NULL COMMENT '操作人',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `remark` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY idx_material (`material_id`),
  KEY idx_ref (`ref_type`,`ref_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='库存变动流水';
